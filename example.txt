const generateToken = (user)=>{
    return jwt.sign({_id:user._id, role:user.role}, secret)
}

const validateToken = (req, res, next)=>{
    try {
        const token =  req.header("Authorization")
        if(!token){
            return res.status(400).json({message:"Authentication failed"})
        }
        const decode = jwt.verify(token, secret)
        req.user = decode
        next()
    } catch (error) {
        return res.status(400).json({message:"Authorization failed"})
        
    }
}

const checkRole =(role)=>(req, res, next)=>{
    try {
        if(req.user.role !== role){
            return res.status(403).json({message:"Access denied"})
        }
        next()
    } catch (error) {
        return res.status(400).json({message:error.message})
    }
}

import React, { useState } from 'react'

function AddAppointment({ addAppointment }) {
    const [formData, setFormData] = useState({
        patientName: '',
        doctorName: '',
        department: '',
        appointmentDate: '',
        timeSlot: '',
        totalFees: '',
        status: 'Scheduled'
    })
    const [errorMessage, setErrorMessage] = useState({})
    const [successMessage, setSuccessMessage] = useState(null)

    const handleInput = (e) => {
        setFormData(prev => ({ ...prev, [e.target.name]: e.target.value }))
    }

    const hadnleSubmit = async () => {
        const errors = {}

        if (!formData.patientName) {
            errors.patientName = 'Patient Name is required'
        }

        if (!formData.doctorName) {
            errors.doctorName = 'Doctor Name is required'
        }

        if (!formData.department) {
            errors.department = 'Department is required'
        }

        if (!formData.appointmentDate) {
            errors.appointmentDate = 'Appointment Date is required'
        }

        if (!formData.timeSlot) {
            errors.timeSlot = 'Appointment Time is required'
        }

        if (!formData.totalFees) {
            errors.totalFees = 'Total fees is required'
        }

        if (!formData.patientName ||
            !formData.doctorName ||
            !formData.department ||
            !formData.appointmentDate ||
            !formData.timeSlot ||
            !formData.totalFees) {
            errors.global = 'Please fill all required fields.'
        }

        if (Object.keys(errors).length > 0) {
            setErrorMessage(errors)
            return
        }

        const obj = { ...formData, id: Date.now(), totalFees: Number(formData.totalFees) }
        setFormData({
            patientName: '',
            doctorName: '',
            department: '',
            appointmentDate: '',
            timeSlot: '',
            totalFees: '',
            status: 'Sheduled'
        })

        const { success } = await addAppointment(obj)

        if (success) {
            setSuccessMessage('Added successfully')
        }
        else {
            setSuccessMessage('Something went wrong. Try again.')
        }
    }

    return (
        <div>

            <div>
                <label htmlFor="patientName">Patient Name: </label>
                <input type="text"
                    name='patientName'
                    id='patientName'
                    placeholder='Patient Name'
                    value={formData.patientName}
                    onChange={(e) => handleInput(e)} />
                {errorMessage.patientName && <p>{errorMessage.patientName}</p>}
            </div>

            <div>
                <label htmlFor="doctorName">Doctor Name: </label>
                <input type="text"
                    name='doctorName'
                    id='doctorName'
                    placeholder='Doctor Name'
                    value={formData.doctorName}
                    onChange={(e) => handleInput(e)} />
                {errorMessage.doctorName && <p>{errorMessage.doctorName}</p>}


            </div>

            <div>
                <label htmlFor="department"></label>
                <select name="department" id="department" value={formData.department} onChange={(e) => handleInput(e)}>
                    <option value="">--Select Department--</option>
                    <option value="Cardiology">Cardiology</option>
                    <option value="Neurology">NeuroLogy</option>
                    <option value="Orthopedics">Orthopedics</option>
                    <option value="General">General</option>
                </select>
                {errorMessage.department && <p>{errorMessage.department}</p>}


            </div>

            <div>
                <label htmlFor="appointmentDate">Appointment Date: </label>
                <input type="date"
                    name='appointmentDate'
                    id='appointmentDate'
                    value={formData.appointmentDate}
                    onChange={(e) => handleInput(e)} />
                {errorMessage.appointmentDate && <p>{errorMessage.appointmentDate}</p>}
            </div>


            <div>
                <label htmlFor="timeSlot">Appointment Time: </label>
                <input type="text"
                    name='timeSlot'
                    id='timeSlot'
                    placeholder='Time Slot'
                    value={formData.timeSlot}
                    onChange={(e) => handleInput(e)} />
                {errorMessage.timeSlot && <p>{errorMessage.timeSlot}</p>}

            </div>

            <div>
                <label htmlFor="totalFees">Total Fees: </label>
                <input type="number"
                    name='totalFees'
                    id='totalFees'
                    placeholder='Total Fees'
                    value={formData.totalFees}
                    onChange={(e) => handleInput(e)} />
                {errorMessage.totalFees && <p>{errorMessage.totalFees}</p>}

            </div>

            <button onClick={() => hadnleSubmit()} >Add Appointment</button>
            {errorMessage.global && <p>{errorMessage.global}</p>}


            {successMessage && <p>{successMessage}</p>}
        </div>
    )
}

export default AddAppointment




import React, { useEffect, useState } from 'react'
import { useNavigate } from 'react-router-dom'

function AppointmentList({ appointments, deleteAppointment, getAppointments }) {
  const [searchValue, setSearchValue] = useState('')
  const [sortValue, setSortValue] = useState(null)
  const navigate = useNavigate()


  let filteredData = appointments.filter(itr => itr.patientName.toLowerCase().includes(searchValue.toLowerCase()) ||
    itr.doctorName.toLowerCase().includes(searchValue.toLowerCase()))

  if (sortValue == "date-asc") {
    filteredData.sort((a, b) => new Date(a.appointmentDate) - new Date(b.appointmentDate))
  }

  if (sortValue == "date-desc") {
    filteredData.sort((a, b) => new Date(b.appointmentDate) - new Date(a.appointmentDate))
  }

  if (sortValue == "fees-asc") {
    filteredData.sort((a, b) => a.totalFees - b.totalFees)
  }

  if (sortValue == "fees-desc") {
    filteredData.sort((a, b) => b.totalFees - a.totalFees)
  }

  if (filteredData.length == 0) {
    return <p>No appointments available</p>
  }
  // useEffect(() => {
  //   getAppointments()
  // }, [])


  return (
    <div>

      <select onChange={(e) => setSortValue(e.target.value)} >
        <option value="date-asc">Sort by Date (oldest first)</option>
        <option value="date-desc">Sort by Date (newest first)</option>
        <option value="fees-asc">Sort by Total Fees (low to high)</option>
        <option value="fees-desc">Sort by Total Fees (High to low)</option>
      </select>

      <label htmlFor="search"></label>
      <input type="text"
        name='search'
        id='search'
        placeholder='Search by Patient Name or Doctor Name'
        onChange={(e) => setSearchValue(e.target.value)} />

      <h1>Appointments</h1>

      
      <table>
        <thead>
          <tr>
            <th>Patient Name</th>
            <th>Doctor Name</th>
            <th>Department</th>
            <th>Appointment Date</th>
            <th>Time Slot</th>
            <th>Status</th>
            <th>Actions</th>
          </tr>
        </thead>

        <tbody>
          {filteredData.length > 0 ? filteredData.map(appointment => {
            return (
              <tr>
                <td>{appointment.patientName}</td>
                <td>{appointment.doctorName}</td>
                <td>{appointment.department}</td>
                <td>{appointment.appointmentDate}</td>
                <td>{appointment.timeSlot}</td>
                <td>{appointment.totalFees}</td>
                <td>{appointment.status}</td>
                <td><button onClick={()=> navigate(`/appointments/${appointment.id}`)} >View</button></td>
                <td><button onClick={() => deleteAppointment(appointment.id)} >Delete</button></td>
              </tr>
            )
          }) : searchValue ? "No matching apointments found" : "No appointments available"}
        </tbody>
      </table>

      <p>Total Appointments: {appointments.length}</p>

    </div>
  )
}

export default AppointmentList
